# eut_speech_audio_processing — ARM64 / Jetson Thor Python deps
# Installed inside the container on top of EutRobAIDockers ARM base
# (which already provides torch 2.8.0a0+nv25.8 + CUDA 13.0 via the NVIDIA JetPack 7 image).
#
# DO NOT pin torch here — it comes from the base image and is already CUDA-enabled.
# numpy 1.26.4, numba 0.59.1, librosa 0.11.0 are also pre-installed in the base image.

# ── Audio capture ─────────────────────────────────────────────────────────────
# runtime: needs portaudio19-dev installed via apt (see Dockerfile.arm)
sounddevice>=0.4.6

# ── Audio / signal processing ────────────────────────────────────────────────
librosa>=0.10.0,<0.12          # 0.11.0 already in base; keep range for flexibility
numpy>=1.26,<2                 # 1.26.4 in base; keep <2 for broad compat
numba>=0.59,<0.65              # 0.59.1 in base; do NOT pin 0.63.1 — LLVM may conflict on ARM

# ── ASR / Whisper ─────────────────────────────────────────────────────────────
# CTranslate2 ships ARM64 wheels; faster-whisper uses it for optimized Whisper inference
faster-whisper>=1.0.0

# ── Wake word ─────────────────────────────────────────────────────────────────
openwakeword>=0.4.0

# onnxruntime (CPU variant) — no onnxruntime-gpu pip wheel available for Jetson ARM via PyPI.
# OpenWakeWord uses ONNX for its models; CPU inference is sufficient for wake word detection.
# For full TensorRT acceleration, models can be exported separately (out of scope here).
onnxruntime>=1.16.0

# ── torchaudio ────────────────────────────────────────────────────────────────
# NOT installed via pip here. PyPI torchaudio 2.8.0 (manylinux_2_28 aarch64)
# links against PyPI torch 2.8.0 stable and fails with:
#   OSError: _torchaudio.abi3.so: undefined symbol: torch_library_impl
# when loaded against NVIDIA's custom torch 2.8.0a0+nv25.8.
# torchaudio is built from source in Dockerfile.arm (see TORCHAUDIO_TAG ARG)
# using --no-build-isolation so it links against the installed NVIDIA torch.

# ── Speaker diarization ───────────────────────────────────────────────────────
diart>=0.9.2
pyannote.core>=4.5
rx>=3.2

# ── HuggingFace ───────────────────────────────────────────────────────────────
# HF_TOKEN env var required at runtime for Pyannote model download.
huggingface_hub==0.21

# ── Database ──────────────────────────────────────────────────────────────────
pymongo>=4.5

# ── ROS build helpers ─────────────────────────────────────────────────────────
lark>=1.3.1                    # required by hri_msgs build
# empy==4.2 is intentionally NOT listed here — it is force-installed via
# --ignore-installed in Dockerfile.arm before this requirements file is read,
# because empy 3.3.4 ships as a distutils package in the base image and cannot
# be uninstalled by pip through the normal mechanism.
catkin_pkg

# ── Coverage / testing (kept for parity with x86) ────────────────────────────
coverage>=7.0
